// Main application class (contains main method).
// Used to test the system functionalities through a console menu.

import java.util.Scanner;

public class RestaurantApp {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        Restaurant res = new Restaurant("KSU Restaurant", 100, 100, 200);

        res.addPerson(new Waiter("Fahad", "W001", 1500));
        res.addPerson(new Customer("Ali", "CU001"));
        res.addDishToMenu(new Dish("Burger", 25));
        res.addDishToMenu(new Dish("Pasta", 30));
        res.addDishToMenu(new Dish("Salad", 18));

        int choice;
        do {
            System.out.println("\n=== " + res.getName() + " Menu ===");
            System.out.println("1) Add Customer");
            System.out.println("2) Add Waiter");
            System.out.println("3) Add Dish to Menu");
            System.out.println("4) List Menu");
            System.out.println("5) Create Order");
            System.out.println("6) List Orders");
            System.out.println("7) List People");
            System.out.println("8) Compute Payroll");
            System.out.println("9) Remove Person by ID");
            System.out.println("10) Remove Dish by name");
            System.out.println("11) Remove Order by #");
            System.out.println("0) Exit");
            System.out.print("Choose: ");
            choice = input.nextInt();
            input.nextLine();

            if (choice == 1) {
                System.out.print("Name: ");
                String name = input.nextLine();
                System.out.print("ID: ");
                String id = input.nextLine();
                boolean ok = res.addPerson(new Customer(name, id));
                System.out.println(ok ? "Added." : "Failed.");
            }

            else if (choice == 2) {
                System.out.print("Name: ");
                String name = input.nextLine();
                System.out.print("ID: ");
                String id = input.nextLine();
                System.out.print("Base salary: ");
                double base = input.nextDouble();
                input.nextLine();
                boolean ok = res.addPerson(new Waiter(name, id, base));
                System.out.println(ok ? "Added." : "Failed.");
            }

            else if (choice == 3) {
                System.out.print("Dish name: ");
                String name = input.nextLine();
                System.out.print("Price: ");
                double price = input.nextDouble();
                input.nextLine();
                boolean ok = res.addDishToMenu(new Dish(name, price));
                System.out.println(ok ? "Added." : "Failed.");
            }

            else if (choice == 4) {
                res.listMenu();
            }

            else if (choice == 5) {
                System.out.print("Customer ID: ");
                String custId = input.nextLine();
                Person p = res.getPersonById(custId);
                if (!(p instanceof Customer)) {
                    System.out.println("Customer not found.");
                    continue;
                }
                Customer customer = (Customer) p;

                System.out.print("Waiter ID (or leave empty): ");
                String waiterId = input.nextLine();
                Waiter waiter = null;
                if (!waiterId.isBlank()) {
                    Person w = res.getPersonById(waiterId);
                    if (w instanceof Waiter) waiter = (Waiter) w;
                    else System.out.println("Waiter not found, continuing without waiter.");
                }

                Order order = new Order(customer, waiter, 20);
                while (true) {
                    res.listMenu();
                    System.out.print("Add dish by name (or 'done'): ");
                    String dn = input.nextLine();
                    if (dn.equalsIgnoreCase("done")) break;
                    Dish d = res.getDishByName(dn);
                    if (d == null) System.out.println("Not in menu.");
                    else {
                        if (!order.addDish(d)) {
                            System.out.println("Order is full.");
                            break;
                        }
                        if (waiter != null) waiter.addTip(2);
                    }
                }

                if (order.size() == 0) {
                    System.out.println("Empty order discarded.");
                } else {
                    boolean ok = res.addOrder(order);
                    System.out.println(ok ? "Order placed. Total = " + order.getTotal() : "Failed to add order.");
                }
            }

            else if (choice == 6) {
                res.listOrders();
            }

            else if (choice == 7) {
                res.listPeople();
            }

            else if (choice == 8) {
                System.out.println("Total payroll: " + res.computeTotalPayroll());
            }

            else if (choice == 9) {
                System.out.print("Person ID to remove: ");
                String id = input.nextLine();
                boolean ok = res.removePersonById(id);
                System.out.println(ok ? "Removed." : "Not found.");
            }

            else if (choice == 10) {
                System.out.print("Dish name to remove: ");
                String name = input.nextLine();
                boolean ok = res.removeDishByName(name);
                System.out.println(ok ? "Removed." : "Not found.");
            }

            else if (choice == 11) {
                System.out.print("Order # to remove: ");
                int idx = input.nextInt();
                input.nextLine();
                boolean ok = res.removeOrderAt(idx);
                System.out.println(ok ? "Removed." : "Invalid order #.");
            }

        } while (choice != 0);

        input.close();
        System.out.println("Bye 👋");
    }
}
